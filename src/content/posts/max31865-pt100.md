# STM32F103C8T6 + MAX31865 读取PT100温度

今天来教大家如何使用 STM32F103C8T6 通过 MAX31865 模块读取 PT100 铂电阻温度传感器~

## 硬件准备 {#准备工作}

### 1. 所需硬件
- STM32F103C8T6 开发板
- MAX31865 模块
- PT100 铂电阻传感器
- 杜邦线若干

## 工作原理 {#代码详解}

### MAX31865 简介

MAX31865 是专门用于 PT100/PT1000 铂电阻温度传感器的数字转换芯片，主要特点：
- 内置 ADC 转换
- 支持 2/3/4 线制
- 故障检测
- SPI 接口

### 温度计算

使用 Callendar-Van Dusen 公式计算温度：
- **正温区**：使用简化公式
- **负温区**：使用 IEC751 多项式拟合

### max31865.h

```C
#ifndef __MAX31865_H
#define __MAX31865_H

#include "stm32f10x.h"
#include "math.h"
#include <stddef.h>

/************************** 硬件配置宏 **************************/
#define MAX31865_PORT             GPIOA
#define MAX31865_PIN_SDO          GPIO_Pin_0
#define MAX31865_PIN_CS           GPIO_Pin_1
#define MAX31865_PIN_SCLK         GPIO_Pin_2
#define MAX31865_PIN_SDI          GPIO_Pin_3
#define MAX31865_PIN_DRDY         GPIO_Pin_15

#define RTD_TYPE_PT100            1
#define RTD_WIRE_3LINE            1
#define RTD_RREF_VALUE            400.0f
#define RTD_R0_VALUE              100.0f
#define RTD_FILTER_50HZ           1

/************************** 驱动状态枚举 **************************/
typedef enum
{
    MAX31865_OK       = 0x00,
    MAX31865_ERROR    = 0x01,
    MAX31865_COMM_ERR = 0x02,
    MAX31865_RTD_OPEN = 0x03,
    MAX31865_RTD_SHORT= 0x04,
    MAX31865_PARAM_ERR= 0x05
}MAX31865_StatusTypeDef;

/************************** 对外接口函数 **************************/
MAX31865_StatusTypeDef MAX31865_Init(void);
MAX31865_StatusTypeDef MAX31865_GetTemperature(float *temp);
MAX31865_StatusTypeDef MAX31865_SoftReset(void);
uint8_t MAX31865_ReadFaultStatus(void);
MAX31865_StatusTypeDef MAX31865_SetConvertMode(uint8_t autoMode);

#endif
```

### max31865.c

```C
#include "max31865.h"

/************************** 底层引脚操作 **************************/
#define CS_LOW()     GPIO_ResetBits(MAX31865_PORT, MAX31865_PIN_CS)
#define CS_HIGH()   GPIO_SetBits(MAX31865_PORT, MAX31865_PIN_CS)
#define SCLK_LOW()  GPIO_ResetBits(MAX31865_PORT, MAX31865_PIN_SCLK)
#define SCLK_HIGH() GPIO_SetBits(MAX31865_PORT, MAX31865_PIN_SCLK)
#define SDI_LOW()   GPIO_ResetBits(MAX31865_PORT, MAX31865_PIN_SDI)
#define SDI_HIGH()  GPIO_SetBits(MAX31865_PORT, MAX31865_PIN_SDI)
#define SDO_READ()  GPIO_ReadInputDataBit(MAX31865_PORT, MAX31865_PIN_SDO)

/************************** 寄存器定义 **************************/
#define MAX31865_REG_CONFIG        0x00
#define MAX31865_REG_RTD_MSB       0x01
#define MAX31865_REG_RTD_LSB       0x02
#define MAX31865_REG_FAULT         0x07
#define MAX31865_WRITE_MASK        0x80

#define CFG_VBIAS_ON               (1<<7)
#define CFG_AUTO_CONVERT           (1<<6)
#define CFG_3WIRE_MODE             (1<<5)
#define CFG_FILTER_50HZ            (1<<0)

/************************** SPI读写函数 **************************/
static void MAX31865_WriteByte(uint8_t data)
{
    for(uint8_t i = 0; i < 8; i++)
    {
        SCLK_LOW();
        if(data & 0x80) SDI_HIGH();
        else SDI_LOW();
        SCLK_HIGH();
        data <<= 1;
    }
}

static uint8_t MAX31865_ReadByte(void)
{
    uint8_t data = 0;
    for(uint8_t i = 0; i < 8; i++)
    {
        SCLK_LOW();
        data <<= 1;
        SCLK_HIGH();
        if(SDO_READ()) data |= 0x01;
    }
    return data;
}

/************************** 寄存器操作 **************************/
static MAX31865_StatusTypeDef MAX31865_WriteReg(uint8_t addr, uint8_t data)
{
    CS_LOW();
    MAX31865_WriteByte(addr | MAX31865_WRITE_MASK);
    MAX31865_WriteByte(data);
    CS_HIGH();
    return MAX31865_OK;
}

static MAX31865_StatusTypeDef MAX31865_ReadReg(uint8_t addr, uint8_t *data)
{
    if(data == NULL) return MAX31865_PARAM_ERR;
    CS_LOW();
    MAX31865_WriteByte(addr);
    *data = MAX31865_ReadByte();
    CS_HIGH();
    return MAX31865_OK;
}

/************************** 初始化 **************************/
MAX31865_StatusTypeDef MAX31865_Init(void)
{
    MAX31865_StatusTypeDef status;
    uint8_t cfgValue = 0;

    // GPIO初始化（略）
    // 软件复位
    status = MAX31865_SoftReset();
    if(status != MAX31865_OK) return status;

    // 组装配置
    cfgValue |= CFG_VBIAS_ON;
    cfgValue |= CFG_AUTO_CONVERT;
    #if RTD_WIRE_3LINE
    cfgValue |= CFG_3WIRE_MODE;
    #endif
    #if RTD_FILTER_50HZ
    cfgValue |= CFG_FILTER_50HZ;
    #endif

    return MAX31865_WriteReg(MAX31865_REG_CONFIG, cfgValue);
}

/************************** 获取温度 **************************/
MAX31865_StatusTypeDef MAX31865_GetTemperature(float *temp)
{
    if(temp == NULL) return MAX31865_PARAM_ERR;

    uint8_t regH, regL, fault;
    uint16_t adcData;
    float Rt, temperature;

    // 读故障状态
    fault = MAX31865_ReadFaultStatus();
    if(fault & 0x04) return MAX31865_RTD_OPEN;
    if(fault & 0x02) return MAX31865_RTD_SHORT;

    // 读RTD数据
    MAX31865_ReadReg(MAX31865_REG_RTD_MSB, &regH);
    MAX31865_ReadReg(MAX31865_REG_RTD_LSB, &regL);

    // 计算温度
    adcData = ((uint16_t)regH << 8 | regL) >> 1;
    if(adcData == 0 || adcData >= 32767) return MAX31865_ERROR;

    Rt = (float)adcData / 32768.0f * RTD_RREF_VALUE;

    // Callendar-Van Dusen公式
    const float a = 3.9083e-3f;
    const float b = -5.775e-7f;

    float Z1 = -a;
    float Z2 = a * a - 4 * b;
    float Z3 = 4 * b / RTD_R0_VALUE;
    float Z4 = 2 * b;
    temperature = Z2 + Z3 * Rt;
    temperature = (sqrtf(temperature) + Z1) / Z4;

    if(temperature >= 0.0f)
    {
        *temp = temperature;
        return MAX31865_OK;
    }

    // 负温多项式
    float rpoly = Rt;
    temperature = -242.02f;
    temperature += 2.2228f * rpoly;
    rpoly *= Rt;
    temperature += 2.5859e-3f * rpoly;
    rpoly *= Rt;
    temperature -= 4.8260e-6f * rpoly;

    *temp = temperature;
    return MAX31865_OK;
}

MAX31865_StatusTypeDef MAX31865_SoftReset(void)
{
    return MAX31865_WriteReg(MAX31865_REG_CONFIG, 0x80);
}

uint8_t MAX31865_ReadFaultStatus(void)
{
    uint8_t fault = 0;
    MAX31865_ReadReg(MAX31865_REG_FAULT, &fault);
    return fault;
}
```

### main.c

```C
#include "max31865.h"

int main(void)
{
    float temp;
    MAX31865_StatusTypeDef status;

    status = MAX31865_Init();
    if(status != MAX31865_OK)
    {
        while(1);
    }

    while(1)
    {
        status = MAX31865_GetTemperature(&temp);
        if(status == MAX31865_OK)
        {
            // 温度有效
        }

        MAX31865_Delay(0xFFFFF);
    }
}
```

### MAX31865.c

```C
#include "MAX31865.h"

void MAX31865_Init(void)
{
    GPIO_InitTypeDef GPIO_InitStruct = {0};

    __HAL_RCC_GPIOA_CLK_ENABLE();

    GPIO_InitStruct.Pin = MAX31865_CS | MAX31865_SCLK | MAX31865_SDI;
    GPIO_InitStruct.Mode = GPIO_MODE_OUTPUT_PP;
    GPIO_InitStruct.Pull = GPIO_NOPULL;
    GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_HIGH;
    HAL_GPIO_Init(MAX31865_CONTROL_PORT, &GPIO_InitStruct);

    GPIO_InitStruct.Pin = MAX31865_SDO;
    GPIO_InitStruct.Mode = GPIO_MODE_INPUT;
    HAL_GPIO_Init(MAX31865_CONTROL_PORT, &GPIO_InitStruct);

    MAX31865_CS_SET;
    MAX31865_SCLK_SET;
}

void MAX31865_Write(unsigned char addr, unsigned char data)
{
    unsigned char i;

    MAX31865_CS_CLR;
    for(i = 0; i < 8; i++)
    {
        MAX31865_SCLK_CLR;
        if(addr & 0x80) MAX31865_SDI_SET;
        else MAX31865_SDI_CLR;
        MAX31865_SCLK_SET;
        addr <<= 1;
    }
    for(i = 0; i < 8; i++)
    {
        MAX31865_SCLK_CLR;
        if(data & 0x80) MAX31865_SDI_SET;
        else MAX31865_SDI_CLR;
        MAX31865_SCLK_SET;
        data <<= 1;
    }
    MAX31865_CS_SET;
}

unsigned char MAX31865_Read(unsigned char addr)
{
    unsigned char i;
    unsigned char data = 0;

    MAX31865_CS_CLR;
    for(i = 0; i < 8; i++)
    {
        MAX31865_SCLK_CLR;
        if(addr & 0x80) MAX31865_SDI_SET;
        else MAX31865_SDI_CLR;
        MAX31865_SCLK_SET;
        addr <<= 1;
    }
    for(i = 0; i < 8; i++)
    {
        MAX31865_SCLK_CLR;
        data <<= 1;
        MAX31865_SCLK_SET;
        if(MAX31865_SDO_READ) data |= 0x01;
    }
    MAX31865_CS_SET;
    return data;
}

float MAX31865_GetTemp(void)
{
    unsigned int data;
    float Rt;
    float Rt0 = 100;
    float Z1, Z2, Z3, Z4, temp;
    float a = 3.9083e-3;
    float b = -5.775e-7;
    float rpoly;

    data = MAX31865_Read(0x01) << 8;
    data |= MAX31865_Read(0x02);
    data >>= 1;

    Rt = (float)data / 32768.0 * RREF;

    Z1 = -a;
    Z2 = a * a - 4 * b;
    Z3 = 4 * b / Rt0;
    Z4 = 2 * b;
    temp = Z2 + Z3 * Rt;
    temp = (sqrt(temp) + Z1) / Z4;

    if(temp >= 0)
    {
        return temp;
    }

    rpoly = Rt;
    temp = -242.02;
    temp += 2.2228 * rpoly;
    rpoly *= Rt;
    temp += 2.5859e-3 * rpoly;
    rpoly *= Rt;
    temp -= 4.8260e-6 * rpoly;
    rpoly *= Rt;
    temp -= 2.8183e-8 * rpoly;
    rpoly *= Rt;
    temp += 1.5243e-10 * rpoly;

    return temp;
}
```
